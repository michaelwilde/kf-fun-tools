Python: Running Discovery with Technical Context

import requests

payload = {
    "metrics": {
        "current_mttr": "2.5 hours",
        "incident_frequency": "15 per week",
        "on_call_burnout": "High - team turnover"
    },
    "pain": "Engineers spend too much time on incident response instead of building",
    "competition": "PagerDuty + manual response",
    "decision_criteria": [
        "Must reduce MTTR by 50%",
        "Must integrate with our Slack and PagerDuty workflows"
    ]
}

response = requests.post(
    "https://resolve.ai/sales/v1/[your-name]/client/discovery", 
    json=payload,
    headers={"Technical-Context": "Kubernetes, AWS, PagerDuty, Slack stack"}
)

print(response.json())
