Python: Running Discovery with Technical Context

import requests

payload = {
    "metrics": {
        "current_uptime": "99.95%",
        "mttr": "45 minutes",
        "alert_noise": "70% false positives"
    },
    "pain": "Engineers spend too much time troubleshooting instead of building",
    "competition": "New Relic",
    "decision_criteria": [
        "Must reduce alert noise by 50%",
        "Must integrate with our CI/CD pipeline"
    ]
}

response = requests.post(
    "https://kloudfuse.com/sales/v1/[your-name]/client/discovery", 
    json=payload,
    headers={"Technical-Context": "Kubernetes, Istio, Prometheus stack"}
)

print(response.json())
